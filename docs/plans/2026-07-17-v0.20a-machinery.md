- [ ] **Step 2: Implement `board/src/lib/bundleState.ts`**

```ts
// The durable bundle state (spec S3): validation status keys everything the
// verdict used to key. Legacy verdict.json still displays (read-only) but no
// new verdicts are ever emitted. Mirrors nothing server-side — board.py keeps
// injecting manifest.validation and verdict as-is.
import type { ResultsBundle } from "./types";

export interface BundleState {
  kind: "validated" | "deviations" | "unvalidated" | "retrofit";
  validation: string | null; // raw validation status, null when absent
  legacyVerdict: "accepted" | "changes-requested" | null;
}

export function bundleState(b: ResultsBundle): BundleState {
  const v = b.manifest?.validation?.status ?? null;
  const legacy = b.verdict?.status ?? null;
  const kind =
    v === "conforms" || v === "conforms-with-amendments"
      ? "validated"
      : v === "deviations-found"
        ? "deviations"
        : v === "not-applicable"
          ? "retrofit"
          : "unvalidated"; // unverifiable | skipped | absent
  return { kind, validation: v, legacyVerdict: legacy };
}

const BADGES: Record<BundleState["kind"], { label: string; cls: string }> = {
  validated: {
    label: "validated",
    cls: "border-green-200 dark:border-green-900 bg-green-50 dark:bg-green-950 text-green-800 dark:text-green-300",
  },
  deviations: {
    label: "deviations flagged",
    cls: "border-red-200 dark:border-red-900 bg-red-50 dark:bg-red-950 text-red-800 dark:text-red-300",
  },
  unvalidated: {
    label: "unvalidated",
    cls: "border-stone-200 dark:border-stone-700 bg-stone-50 dark:bg-stone-800 text-stone-600 dark:text-stone-300",
  },
  retrofit: {
    label: "retrofit — no plan validation",
    cls: "border-stone-200 dark:border-stone-700 bg-stone-50 dark:bg-stone-800 text-stone-600 dark:text-stone-300",
  },
};

export function bundleStateBadge(b: ResultsBundle): { label: string; cls: string } {
  const s = bundleState(b);
  const base = BADGES[s.kind];
  return s.validation === "conforms-with-amendments"
    ? { ...base, label: "validated — amendments recorded" }
    : base;
}

export function bundleStateMark(b: ResultsBundle): string {
  const s = bundleState(b);
  return s.kind === "validated" ? " ✓" : s.kind === "deviations" ? " ✕" : " ●";
}
```
